246  Debug.Log("Server - Connectevent");

252  byte[] bytes = strm.ReadBytesAsArray(ref readerCtx, strm.Length);
253  string data = Encoding.ASCII.GetString(bytes);
254 
256  var writer = new DataStreamWriter(1024, Allocator.Temp);
257  if (data.Contains("|"))

260

262  var type = Type.GetType(str[1]);
263  dynamic msg = JsonUtility.FromJson(str[0], type);
264 
265  MessagingManager.BroadcastMessage((Message) msg, i);
266  }
267  else if (data.Contains("<Scenario>"))
268  {
270  writer.Write(Encoding.ASCII.GetBytes(data));
271  for (int j = 0; j < m_connections.Length; j++)
272  {
273  m_ServerDriver.Send(m_connections[j], writer);
274  }
275  Debug.Log("Server - Got scenario: " + data);
276  }
277 
279  writer.Dispose();
